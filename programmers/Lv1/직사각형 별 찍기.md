## 풀이
```javascript
process.stdin.setEncoding('utf8');
process.stdin.on('data', data => {
    const n = data.split(" ");
    const a = Number(n[0]), b = Number(n[1]);
    let star = ""
    for (let i = 0; i < b; i++) {
        for (let n = 0; n < a; n++) {
            star += '*'
        }
            star += '\n'
    }
    console.log(star);
});

// a = 별의 개수 / b = 몇 줄
// 빈문자열 star 선언
// 이중반복문
// b미만까지 a미만까지
// a 숫자미만까지 star에 별 추가
// a 반복문 끝나면 줄바꿈 추가
```
